  <script src="https://cdnjs.cloudflare.com/ajax/libs/PapaParse/5.4.1/papaparse.min.js"></script>

  <style>

    body { font-family: sans-serif; text-align: center; margin: 0; background: #f8f8f8; }

    #map { width: 100%; height: 80vh; background: #e0e0e0; }

    .pin { stroke: #000; stroke-width: 0.2px; }

    .legend { margin-top: 1rem; }

    .controls { margin: 1rem; }

  </style>

<body>

  <h1>FaithTech Community Map Generator</h1>

  <div class="controls">

    <input type="file" id="upload" accept=".csv">

    <button onclick="downloadSVG()">Download SVG</button>

    <button onclick="downloadPNG()">Download PNG</button>

  </div>

  <svg id="map"></svg>

  <script>

    const width = 1000, height = 600;

    const svg = d3.select("#map").attr("width", width).attr("height", height);

    const projection = d3.geoMercator().center([-95, 40]).scale(700).translate([width / 2, height / 2]);

    const path = d3.geoPath().projection(projection);


    // Base map (Natural Earth 110m simplified geojson, hosted via CDN)

    d3.json("https://cdn.jsdelivr.net/npm/world-atlas@2/countries-110m.json").then(world => {

      const countries = topojson.feature(world, world.objects.countries).features;

      svg.append("g")

        .selectAll("path")

        .data(countries)

        .join("path")

        .attr("d", path)

        .attr("fill", "#f3f3f3")

        .attr("stroke", "#ccc");

    });


    document.getElementById('upload').addEventListener('change', function(event) {

      const file = event.target.files[0];

      Papa.parse(file, {

        header: true,

        dynamicTyping: true,

        complete: function(results) {

          plotPins(results.data);

        }

      });

    });


    function plotPins(data) {

      svg.selectAll("circle").remove();


      data.forEach(d => {

        if (!d.Latitude || !d.Longitude) return;

        const [x, y] = projection([+d.Longitude, +d.Latitude]);

        let color = d.Status.includes("Current") ? "#F05228" : "#E4E0D8";


        svg.append("circle")

          .attr("cx", x)

          .attr("cy", y)

          .attr("r", 5)

          .attr("fill", color)

          .attr("class", "pin")

          .append("title")

          .text(d.City);

      });

    }


    function downloadSVG() {

      const svgData = new XMLSerializer().serializeToString(svg.node());

      const blob = new Blob([svgData], {type: "image/svg+xml;charset=utf-8"});

      const url = URL.createObjectURL(blob);

      const link = document.createElement("a");

      link.href = url;

      link.download = "faithtech-map.svg";

      link.click();

    }


    function downloadPNG() {

      const canvas = document.createElement("canvas");

      canvas.width = width;

      canvas.height = height;

      const ctx = canvas.getContext("2d");

      const image = new Image();

      const svgData = new XMLSerializer().serializeToString(svg.node());

      const svgBlob = new Blob([svgData], {type: "image/svg+xml;charset=utf-8"});

      const url = URL.createObjectURL(svgBlob);

      image.onload = function() {

        ctx.drawImage(image, 0, 0);

        URL.revokeObjectURL(url);

        const pngURL = canvas.toDataURL("image/png");

        const link = document.createElement("a");

        link.href = pngURL;

        link.download = "faithtech-map.png";

        link.click();

      };

      image.src = url;

    }

  </script>
